function qs(name) { return new URLSearchParams(location.search).get(name) || ''; } function findHubUrl() { return ( document.querySelector('[data-hub]')?.getAttribute('data-hub') || qs('hub') || '../../hub.html' ); } function bindLink(btn, flow, label, targetUrl, reason) { if (!btn) return; btn.addEventListener('click', (ev) => { ev.preventDefault(); const nextUrl = appendFlowParams(targetUrl, flow.ctx); flow.noteRedirect(reason, nextUrl, { nextLabel: label, nextPath: nextUrl }); flow.complete(reason, { nextLabel: label, nextPath: nextUrl }); location.href = nextUrl; }); } async function init() { const flow = initFlowBridge({ pageId: 'parent-summary', defaultStep: 'reviewing' }); await flow.start(); flow.setStep('reviewing', { reviewStartedAt: new Date().toISOString() }); const hubBtn = document.querySelector('[data-action="hub"]') || document.querySelector('[data-role="hub"]'); const finishBtn = document.querySelector('[data-action="finish"]') || document.querySelector('[data-role="finish"]'); const printBtn = document.querySelector('[data-action="print"]') || document.querySelector('[data-role="print"]'); const restartBtn = document.querySelector('[data-action="restart"]') || document.querySelector('[data-role="restart"]'); bindLink(hubBtn, flow, 'กลับ HUB', findHubUrl(), 'parent-summary-hub'); if (finishBtn) { const finishUrl = finishBtn.getAttribute('data-next') || findHubUrl(); bindLink(finishBtn, flow, 'Finish', finishUrl, 'parent-summary-finish'); } if (restartBtn) { const restartUrl = restartBtn.getAttribute('data-next') || '../game/goodjunk-vr.html'; bindLink(restartBtn, flow, 'Restart Game', restartUrl, 'parent-summary-restart'); } if (printBtn) { printBtn.addEventListener('click', () => { flow.persist({ flowStep: 'reviewing', printClickedAt: new Date().toISOString() }); flow.pushEvent('parent-summary-print', { pageId: 'parent-summary' }).catch(() => {}); window.print(); }); } window.addEventListener('beforeprint', () => { flow.persist({ flowStep: 'reviewing', beforePrintAt: new Date().toISOString() }); }); } document.addEventListener('DOMContentLoaded', init);
