  function stepDone(step) {

  async function flushOutbox() {
    if(!ENDPOINT||!navigator.onLine)return;
    const outbox=readJson(STORAGE.outbox,[]);if(!outbox.length)return;
    const remaining=[];

      try{
        await fetch(ENDPOINT,{method:'POST',mode:'no-cors',cache:'no-store',keepalive:true,headers:{'Content-Type':'text/plain;charset=utf-8'},body:JSON.stringify(item.payload)});
      }catch(error){item.tries=Number(item.tries||0)+1;remaining.push(item);}
    }
    writeJson(STORAGE.outbox,remaining);
    if(el.deliveryText&&el.summaryOverlay.classList.contains('show'))el.deliveryText.textContent=remaining.length?'เก็บไว้ใน Outbox และจะส่งซ้ำเมื่อออนไลน์':'ส่งผลออกจาก Outbox แล้ว';
